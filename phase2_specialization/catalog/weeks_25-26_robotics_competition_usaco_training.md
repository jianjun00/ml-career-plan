class USACOProblemSolver:
    def __init__(self):
        self.algorithms = {}
        self.strategies = {}
        self.solutions = {}
        
    def problem_analysis(self, problem_statement):
        """Analyze USACO problems systematically"""
        analysis_steps = {
            'constraints': 'Identify input limits and constraints',
            'requirements': 'Understand expected output format',
            'complexity': 'Determine required time/space complexity',
            'edge_cases': 'Identify special cases and boundary conditions'
        }
        
        # Problem categorization
        problem_types = {
            'simulation': 'Direct simulation of described process',
            'greedy': 'Making locally optimal choices',
            'dynamic_programming': 'Optimal substructure and overlapping subproblems',
            'graph': 'Graph traversal and algorithms',
            'mathematical': 'Number theory, geometry, combinatorics'
        }
        
        return analysis_steps, problem_types
        
    def algorithm_selection(self, problem_type, constraints):
        """Select appropriate algorithm based on problem characteristics"""
        algorithm_map = {
            'simulation': ['direct_simulation', 'state_machine'],
            'greedy': ['sorting', 'priority_queue', 'two_pointers'],
            'dynamic_programming': ['memoization', 'bottom_up', 'optimization'],
            'graph': ['bfs', 'dfs', 'dijkstra', 'floyd_warshall'],
            'mathematical': ['number_theory', 'geometry', 'combinatorics']
        }
        
        return algorithm_map.get(problem_type, [])
        
    def solution_implementation(self, algorithm, constraints):
        """Implement optimized solution"""
        implementation_steps = {
            'input_handling': 'Efficient input reading and parsing',
            'core_logic': 'Main algorithm implementation',
            'output_formatting': 'Correct output format',
            'optimization': 'Memory and time optimization'
        }
        
        # Common optimization techniques
        optimizations = {
            'memory': 'Use appropriate data structures',
            'time': 'Eliminate unnecessary computations',
            'io': 'Fast input/output methods',
            'precomputation': 'Preprocess reusable data'
        }
        
        return implementation_steps, optimizations
